React Accessible Shuttle is a low-level, flexible shuttle component written in React.

## Installation

First, make sure to have `react` and `react-dom` installed:

```bash
npm i react react-dom
```

Then, install `react-accessible-shuttle`:

```bash
npm i react-accessible-shuttle
```

## Usage

Play with the examples on [CodeSandbox](https://codesandbox.io/s/react-accessible-shuttle-nd4d8).
See different implementations on codesandbox such as filtering shuttles, drag-and-drop support,
virtualization, and much more.

The below example demonstrates the basic usage of `react-accessible-shuttle`.

```jsx
import React from 'react';
import ReactDOM from 'react-dom';

import { Shuttle, useShuttleState } from 'react-accessible-shuttle';
import 'react-accessible-shuttle/css/shuttle.css';

function App() {
    const shuttle = useShuttleState({
        source: ['a', 'b', 'c'],
        target: ['d', 'e', 'f'],
    });

    return (
        <Shuttle {...shuttle}>
            <Shuttle.Container>
                {({ source, selected }, getItemProps) =>
                    source.map((item, index) => (
                        <Shuttle.Item
                            {...getItemProps(index)}
                            key={item}
                            value={item}
                            selected={selected.source.has(index)}>
                            {item}
                        </Shuttle.Item>
                    ))
                }
            </Shuttle.Container>
            <Shuttle.Controls />
            <Shuttle.Container>
                {({ target, selected }, getItemProps) =>
                    target.map((item, index) => (
                        <Shuttle.Item
                            {...getItemProps(index)}
                            key={item}
                            value={item}
                            selected={selected.target.has(index)}>
                            {item}
                        </Shuttle.Item>
                    ))
                }
            </Shuttle.Container>
        </Shuttle>
    );
}

ReactDOM.render(<App />, document.getElementById('app'));
```

## CDN

You can also use the UMD version of React Accessible Shuttle.
See an example on [JSBin](https://jsbin.com/mivotiy/edit?html,js,output).

```html
<!DOCTYPE html>
<html>
    <head>
        <meta name="description" content="React Accessible Shuttle UMD Example" />
        <meta charset="utf-8" />
        <meta name="viewport" content="width=device-width" />
        <title>JS Bin</title>
        <link
            rel="stylesheet"
            type="text/css"
            href="https://unpkg.com/react-accessible-shuttle@0.3.2/css/shuttle.css"
        />
    </head>
    <body>
        <div id="root"></div>
        <!-- peer dependencies -->
        <script src="https://unpkg.com/react@16.9.0/umd/react.development.js"></script>
        <script src="https://unpkg.com/react-dom@16.9.0/umd/react-dom.development.js"></script>

        <!-- react-accessible-shuttle -->
        <script src="https://unpkg.com/react-accessible-shuttle@0.3.2/dist-browser/index.js"></script>
        <script>
            var Shuttle = ReactShuttle.Shuttle;

            function App() {
                var shuttle = ReactShuttle.useShuttleState({
                    source: ['a', 'b', 'c'],
                    target: ['d', 'e', 'f'],
                });

                return React.createElement(Shuttle, shuttle, [
                    React.createElement(Shuttle.Container, null, function(args, getItemProps) {
                        return args.source.map(function(item, index) {
                            var props = getItemProps(index);

                            props.key = index;
                            props.value = item;

                            return React.createElement(Shuttle.Item, props, item);
                        });
                    }),
                    React.createElement(Shuttle.Controls),
                    React.createElement(Shuttle.Container, null, function(args, getItemProps) {
                        return args.target.map(function(item, index) {
                            var props = getItemProps(index);

                            props.key = index;
                            props.value = item;

                            return React.createElement(Shuttle.Item, props, item);
                        });
                    }),
                ]);
            }

            ReactDOM.render(React.createElement(App), document.getElementById('root'));
        </script>
    </body>
</html>
```
